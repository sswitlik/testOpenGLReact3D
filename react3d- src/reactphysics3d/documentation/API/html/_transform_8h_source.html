156 // Get the OpenGL matrix of the transform
157 inline void Transform::getOpenGLMatrix(decimal* openglMatrix) const {
158  const Matrix3x3& matrix = mOrientation.getMatrix();
159  openglMatrix[0] = matrix[0][0]; openglMatrix[1] = matrix[1][0];
160  openglMatrix[2] = matrix[2][0]; openglMatrix[3] = 0.0;
161  openglMatrix[4] = matrix[0][1]; openglMatrix[5] = matrix[1][1];
162  openglMatrix[6] = matrix[2][1]; openglMatrix[7] = 0.0;
163  openglMatrix[8] = matrix[0][2]; openglMatrix[9] = matrix[1][2];
164  openglMatrix[10] = matrix[2][2]; openglMatrix[11] = 0.0;
165  openglMatrix[12] = mPosition.x; openglMatrix[13] = mPosition.y;
166  openglMatrix[14] = mPosition.z; openglMatrix[15] = 1.0;
167 }

177 inline Transform Transform::interpolateTransforms(const Transform& oldTransform,
178  const Transform& newTransform,
179  decimal interpolationFactor) {
180 
181  Vector3 interPosition = oldTransform.mPosition * (decimal(1.0) - interpolationFactor) +
182  newTransform.mPosition * interpolationFactor;
183 
184  Quaternion interOrientation = Quaternion::slerp(oldTransform.mOrientation,
185  newTransform.mOrientation,

188  return Transform(interPosition, interOrientation);
