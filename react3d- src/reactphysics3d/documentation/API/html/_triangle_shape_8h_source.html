137 // Return the local bounds of the shape in x, y and z directions.
138 // This method is used to compute the AABB of the box
143 inline void TriangleShape::getLocalBounds(Vector3& min, Vector3& max) const {
144 
145  const Vector3 xAxis(mPoints[0].x, mPoints[1].x, mPoints[2].x);
146  const Vector3 yAxis(mPoints[0].y, mPoints[1].y, mPoints[2].y);
147  const Vector3 zAxis(mPoints[0].z, mPoints[1].z, mPoints[2].z);
148  min.setAllValues(xAxis.getMinValue(), yAxis.getMinValue(), zAxis.getMinValue());
149  max.setAllValues(xAxis.getMaxValue(), yAxis.getMaxValue(), zAxis.getMaxValue());
150 
151  min -= Vector3(mMargin, mMargin, mMargin);
152  max += Vector3(mMargin, mMargin, mMargin);
153 }

175 // Update the AABB of a body using its collision shape
181 inline void TriangleShape::computeAABB(AABB& aabb, const Transform& transform) const {
182 
183  const Vector3 worldPoint1 = transform * mPoints[0];
184  const Vector3 worldPoint2 = transform * mPoints[1];
185  const Vector3 worldPoint3 = transform * mPoints[2];
186 
187  const Vector3 xAxis(worldPoint1.x, worldPoint2.x, worldPoint3.x);
188  const Vector3 yAxis(worldPoint1.y, worldPoint2.y, worldPoint3.y);
189  const Vector3 zAxis(worldPoint1.z, worldPoint2.z, worldPoint3.z);
190  aabb.setMin(Vector3(xAxis.getMinValue(), yAxis.getMinValue(), zAxis.getMinValue()));
191  aabb.setMax(Vector3(xAxis.getMaxValue(), yAxis.getMaxValue(), zAxis.getMaxValue()));
192 }
