158 // Return the pair of bodies index
159 inline overlappingpairid OverlappingPair::computeID(ProxyShape* shape1, ProxyShape* shape2) {
160  assert(shape1->mBroadPhaseID >= 0 && shape2->mBroadPhaseID >= 0);
161 
162  // Construct the pair of body index
163  overlappingpairid pairID = shape1->mBroadPhaseID < shape2->mBroadPhaseID ?
164  std::make_pair(shape1->mBroadPhaseID, shape2->mBroadPhaseID) :
165  std::make_pair(shape2->mBroadPhaseID, shape1->mBroadPhaseID);
166  assert(pairID.first != pairID.second);
167  return pairID;
168 }
