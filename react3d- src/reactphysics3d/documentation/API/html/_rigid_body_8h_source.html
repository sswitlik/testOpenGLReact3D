275 inline Matrix3x3 RigidBody::getInertiaTensorWorld() const {
276 
277  // Compute and return the inertia tensor in world coordinates
278  return mTransform.getOrientation().getMatrix() * mInertiaTensorLocal *
279  mTransform.getOrientation().getMatrix().getTranspose();
280 }

292 inline Matrix3x3 RigidBody::getInertiaTensorInverseWorld() const {

294  // TODO : DO NOT RECOMPUTE THE MATRIX MULTIPLICATION EVERY TIME. WE NEED TO STORE THE

296 
297  // Compute and return the inertia tensor in world coordinates

407 inline void RigidBody::applyForceToCenterOfMass(const Vector3& force) {
408 
409  // If it is not a dynamic body, we do nothing
410  if (mType != DYNAMIC) return;
411 
412  // Awake the body if it was sleeping
413  if (mIsSleeping) {
414  setIsSleeping(false);
415  }
416 
417  // Add the force
418  mExternalForce += force;
419 }

432 inline void RigidBody::applyForce(const Vector3& force, const Vector3& point) {
433 
434  // If it is not a dynamic body, we do nothing
435  if (mType != DYNAMIC) return;
436 
437  // Awake the body if it was sleeping
438  if (mIsSleeping) {
439  setIsSleeping(false);
440  }
441 
442  // Add the force and torque
443  mExternalForce += force;
444  mExternalTorque += (point - mCenterOfMassWorld).cross(force);
445 }
