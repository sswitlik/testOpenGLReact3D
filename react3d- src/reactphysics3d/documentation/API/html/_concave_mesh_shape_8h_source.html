208

221 
222  // Get the node data (triangle index and mesh subpart index)

224

226  Vector3 trianglePoints[3];
227  mConcaveMeshShape.getTriangleVerticesWithIndexPointer(data[0], data[1], trianglePoints);
228 
229  // Call the callback to test narrow-phase collision with this triangle
