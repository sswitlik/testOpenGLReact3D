222 // Return the height of a given (x,y) point in the height field
223 inline decimal HeightFieldShape::getHeightAt(int x, int y) const {
224 
225  switch(mHeightDataType) {
226  case HEIGHT_FLOAT_TYPE : return ((float*)mHeightFieldData)[y * mNbColumns + x];
227  case HEIGHT_DOUBLE_TYPE : return ((double*)mHeightFieldData)[y * mNbColumns + x];
228  case HEIGHT_INT_TYPE : return ((int*)mHeightFieldData)[y * mNbColumns + x] * mIntegerHeightScale;
229  default: assert(false); return 0;
230  }
231 }

233 // Return the closest inside integer grid value of a given floating grid value
234 inline int HeightFieldShape::computeIntegerGridValue(decimal value) const {
235  return (value < decimal(0.0)) ? value - decimal(0.5) : value + decimal(0.5);
