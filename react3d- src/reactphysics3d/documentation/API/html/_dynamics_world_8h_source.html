308 // Reset the external force and torque applied to the bodies
309 inline void DynamicsWorld::resetBodiesForceAndTorque() {
310 
311  // For each body of the world
312  std::set<RigidBody*>::iterator it;
313  for (it = mRigidBodies.begin(); it != mRigidBodies.end(); ++it) {
314  (*it)->mExternalForce.setToZero();
315  (*it)->mExternalTorque.setToZero();
316  }
317 }
