180 inline void ConvexMeshShape::computeLocalInertiaTensor(Matrix3x3& tensor, decimal mass) const {
181  decimal factor = (decimal(1.0) / decimal(3.0)) * mass;
182  Vector3 realExtent = decimal(0.5) * (mMaxBounds - mMinBounds);
183  assert(realExtent.x > 0 && realExtent.y > 0 && realExtent.z > 0);
184  decimal xSquare = realExtent.x * realExtent.x;
185  decimal ySquare = realExtent.y * realExtent.y;
186  decimal zSquare = realExtent.z * realExtent.z;
187  tensor.setAllValues(factor * (ySquare + zSquare), 0.0, 0.0,
188  0.0, factor * (xSquare + zSquare), 0.0,
189  0.0, 0.0, factor * (xSquare + ySquare));
190 }

192 // Add a vertex into the convex mesh
196 inline void ConvexMeshShape::addVertex(const Vector3& vertex) {
197 
198  // Add the vertex in to vertices array
199  mVertices.push_back(vertex);
200  mNbVertices++;
201 
202  // Update the bounds of the mesh
203  if (vertex.x * mScaling.x > mMaxBounds.x) mMaxBounds.x = vertex.x * mScaling.x;
204  if (vertex.x * mScaling.x < mMinBounds.x) mMinBounds.x = vertex.x * mScaling.x;
205  if (vertex.y * mScaling.y > mMaxBounds.y) mMaxBounds.y = vertex.y * mScaling.y;
206  if (vertex.y * mScaling.y < mMinBounds.y) mMinBounds.y = vertex.y * mScaling.y;
207  if (vertex.z * mScaling.z > mMaxBounds.z) mMaxBounds.z = vertex.z * mScaling.z;
208  if (vertex.z * mScaling.z < mMinBounds.z) mMinBounds.z = vertex.z * mScaling.z;
209 }

219 inline void ConvexMeshShape::addEdge(uint v1, uint v2) {
220 
221  // If the entry for vertex v1 does not exist in the adjacency list
222  if (mEdgesAdjacencyList.count(v1) == 0) {
223  mEdgesAdjacencyList.insert(std::make_pair(v1, std::set<uint>()));
224  }
225 
226  // If the entry for vertex v2 does not exist in the adjacency list
227  if (mEdgesAdjacencyList.count(v2) == 0) {
228  mEdgesAdjacencyList.insert(std::make_pair(v2, std::set<uint>()));
229  }
230 
231  // Add the edge in the adjacency list
232  mEdgesAdjacencyList[v1].insert(v2);
233  mEdgesAdjacencyList[v2].insert(v1);
234 }
