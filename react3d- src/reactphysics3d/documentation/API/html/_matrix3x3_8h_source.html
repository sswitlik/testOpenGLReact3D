195 // Return the determinant of the matrix
196 inline decimal Matrix3x3::getDeterminant() const {
197 
198  // Compute and return the determinant of the matrix
199  return (mRows[0][0]*(mRows[1][1]*mRows[2][2]-mRows[2][1]*mRows[1][2]) -
200  mRows[0][1]*(mRows[1][0]*mRows[2][2]-mRows[2][0]*mRows[1][2]) +
201  mRows[0][2]*(mRows[1][0]*mRows[2][1]-mRows[2][0]*mRows[1][1]));
202 }
