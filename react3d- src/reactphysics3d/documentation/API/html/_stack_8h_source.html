89 // Push an element into the stack
90 template<typename T, uint capacity>
91 inline void Stack<T, capacity>::push(const T& element) {
92 
93  // If we need to allocate more elements
94  if (mNbElements == mNbAllocatedElements) {
95  T* oldElements = mElements;
96  mNbAllocatedElements *= 2;
97  mElements = (T*) malloc(mNbAllocatedElements * sizeof(T));
98  assert(mElements);
99  memcpy(mElements, oldElements, mNbElements * sizeof(T));
100  if (oldElements != mInitArray) {
101  free(oldElements);
102  }
103  }
104 
105  mElements[mNbElements] = element;
106  mNbElements++;
107 }
