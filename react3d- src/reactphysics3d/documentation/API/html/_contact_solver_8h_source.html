487 // Compute the collision restitution factor from the restitution factor of each body
488 inline decimal ContactSolver::computeMixedRestitutionFactor(RigidBody* body1,
489  RigidBody* body2) const {
490  decimal restitution1 = body1->getMaterial().getBounciness();
491  decimal restitution2 = body2->getMaterial().getBounciness();
492 
493  // Return the largest restitution factor
494  return (restitution1 > restitution2) ? restitution1 : restitution2;
495 }

497 // Compute the mixed friction coefficient from the friction coefficient of each body
498 inline decimal ContactSolver::computeMixedFrictionCoefficient(RigidBody *body1,
499  RigidBody *body2) const {
500  // Use the geometric mean to compute the mixed friction coefficient
501  return sqrt(body1->getMaterial().getFrictionCoefficient() *
502  body2->getMaterial().getFrictionCoefficient());
503 }

505 // Compute th mixed rolling resistance factor between two bodies
506 inline decimal ContactSolver::computeMixedRollingResistance(RigidBody* body1,
507  RigidBody* body2) const {
508  return decimal(0.5f) * (body1->getMaterial().getRollingResistance() + body2->getMaterial().getRollingResistance());
